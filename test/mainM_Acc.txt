clear all
clc
close all
filename = 'data.csv';

T = readtable(filename); %check T.Properties
VariableNames = T.Properties.VariableNames;

Arr = table2array(T);
[m,n] = size(Arr);


figure 
plot(Arr(:,1), Arr(:,4), 'color', 'k','linewidth',2);
grid on
xlabel('Vrijeme [s]'); 
ylabel('a_z [ms^{-2}]'); 
xlim([0 11]);
title('Vertikalno ubrznje dobijeno iz mj\_Data');
latex_fig(10, 3, 1.5);

figure
plot(Arr(:,1), Arr(:,7), 'color', 'k','linewidth',2);
grid on
xlabel('Vrijeme [s]'); 
ylabel('a_z [ms^{-2}]');
xlim([0 11]);
title('Vertikalno ubrznje dobijeno formulom'); 
latex_fig(10, 3, 1.5);

figure
plot(Arr(:,1), Arr(:,4)-Arr(:,7), 'color', 'k','linewidth',2);
grid on
xlabel('Vrijeme [s]'); 
ylabel('\Delta q [ms^{-2}]');
xlim([0 11]);
title('Grafik gre≈°ke');
latex_fig(10, 3, 1.5);
